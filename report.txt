import React from 'react';
import { Card, CardContent, Typography } from '@mui/material';

const StatusCard = ({ id, expectedCre, notReceivedCre }) => {
  let cardColor = 'white'; // Default color

  if (expectedCre.length > 0 && notReceivedCre.length > 0) {
    cardColor = 'red';
  } else if (expectedCre.length > 0) {
    cardColor = 'yellow';
  } else if (notReceivedCre.length > 0) {
    cardColor = 'orange';
  }

  return (
    <Card sx={{ backgroundColor: cardColor, margin: 2, width: '200px', height: '100px' }}>
      <CardContent>
        <Typography variant="h6">Card {id}</Typography>
        <Typography variant="body2">Expected: {expectedCre.length}</Typography>
        <Typography variant="body2">Not Received: {notReceivedCre.length}</Typography>
      </CardContent>
    </Card>
  );
};

export default StatusCard;
