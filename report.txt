import React, { useState } from 'react';
import { Tabs, Tab, Box, Typography, Container } from '@mui/material';

// Composants pour chaque onglet
function Component1() {
  return <Typography variant="h6">Contenu de l'onglet 1</Typography>;
}

function Component2() {
  return <Typography variant="h6">Contenu de l'onglet 2</Typography>;
}

function Component3() {
  return <Typography variant="h6">Contenu de l'onglet 3</Typography>;
}

function Component4() {
  return <Typography variant="h6">Contenu de l'onglet 4</Typography>;
}

// Fonction qui gère l'affichage du contenu en fonction de l'onglet
function TabPanel(props) {
  const { children, value, index, ...other } = props;

  return (
    <div
      role="tabpanel"
      hidden={value !== index}
      id={`tabpanel-${index}`}
      aria-labelledby={`tab-${index}`}
      {...other}
    >
      {value === index && (
        <Box p={3}>
          <Typography>{children}</Typography>
        </Box>
      )}
    </div>
  );
}

// Onglets principaux
const MyTabs = () => {
  const [value, setValue] = useState(0);

  const handleChange = (event, newValue) => {
    setValue(newValue);
  };

  return (
    <Container>
      <Box sx={{ borderBottom: 1, borderColor: 'divider' }}>
        <Tabs value={value} onChange={handleChange} aria-label="basic tabs">
          <Tab label="Onglet 1" />
          <Tab label="Onglet 2" />
          <Tab label="Onglet 3" />
          <Tab label="Onglet 4" />
        </Tabs>
      </Box>

      {/* Rendu des composants selon l'onglet sélectionné */}
      <TabPanel value={value} index={0}>
        <Component1 />
      </TabPanel>
      <TabPanel value={value} index={1}>
        <Component2 />
      </TabPanel>
      <TabPanel value={value} index={2}>
        <Component3 />
      </TabPanel>
      <TabPanel value={value} index={3}>
        <Component4 />
      </TabPanel>
    </Container>
  );
};

export default MyTabs;
