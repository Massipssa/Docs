const handleProjectClick = (projectId) => {
  const selectedProject = filteredProjects.find(project => project.name === projectId);

  console.log('selectedProject', selectedProject);
  console.log('projectId', projectId);

  if (selectedProject) {
    // Ensure creData is an array and critical is a valid field
    const filteredCRE = selectedProject.critical?.filter(cre => {
      // Assuming 'name' exists in the critical array elements and you're filtering by CRE name
      return creData.some(creItem => creItem.name === cre.name);
    }) || [];

    // Update the filtered CREs
    setFilteredCreData(filteredCRE);
  } else {
    console.error('Selected project not found.');
  }
};
