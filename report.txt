import React, { useState } from 'react';
import { Drawer, Box, Button, Typography } from '@mui/material';

function CustomDrawer() {
  const [isDrawerOpen, setIsDrawerOpen] = useState(false);

  const toggleDrawer = (open) => (event) => {
    // Close the drawer if clicking outside or pressing escape
    if (event.type === 'keydown' && (event.key === 'Tab' || event.key === 'Shift')) {
      return;
    }
    setIsDrawerOpen(open);
  };

  return (
    <Box>
      {/* Button to open the drawer */}
      <Button onClick={() => setIsDrawerOpen(true)}>Open Drawer</Button>

      {/* Drawer component */}
      <Drawer
        anchor="right"
        open={isDrawerOpen}
        onClose={toggleDrawer(false)} // Close the drawer when clicking outside
      >
        {/* Drawer Content */}
        <Box sx={{ width: 250, padding: 2 }}>
          <Typography variant="h6">Drawer Content</Typography>
          <Button onClick={() => setIsDrawerOpen(false)}>Close Drawer</Button>
        </Box>
      </Drawer>
    </Box>
  );
}

export default CustomDrawer;
