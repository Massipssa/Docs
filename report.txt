import React from 'react';
import { Grid, Card, CardContent, Typography, IconButton, Tooltip } from '@mui/material';
import CheckCircleIcon from '@mui/icons-material/CheckCircle';
import WarningIcon from '@mui/icons-material/Warning';
import ErrorIcon from '@mui/icons-material/Error';

const projectData = [
  { id: 1, name: "PRJ.M5/M6 Pilo/Gesco", status: "critical" },
  { id: 2, name: "PRJ.M26 - IRB Portability", status: "integrated" },
  { id: 3, name: "PRJ.M9 Telephony", status: "non-critical" },
  // Add more projects as needed
];

const getStatusIcon = (status) => {
  switch (status) {
    case "integrated":
      return <CheckCircleIcon style={{ color: "green" }} />;
    case "non-critical":
      return <WarningIcon style={{ color: "yellow" }} />;
    case "critical":
      return <ErrorIcon style={{ color: "red" }} />;
    default:
      return null;
  }
};

const ProjectsSection = () => {
  return (
    <Grid container spacing={3}>
      {projectData.map((project) => (
        <Grid item xs={12} sm={6} md={4} key={project.id}>
          <Card>
            <CardContent>
              <Grid container alignItems="center">
                <Grid item xs={10}>
                  <Typography variant="h6" noWrap>{project.name}</Typography>
                </Grid>
                <Grid item xs={2}>
                  {getStatusIcon(project.status)}
                </Grid>
              </Grid>
            </CardContent>
          </Card>
        </Grid>
      ))}
    </Grid>
  );
};

export default ProjectsSection;
