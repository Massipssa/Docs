apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: deny-gateway-gravité-to-external-bridge
  namespace: <votre-namespace>
spec:
  podSelector:
    matchLabels:
      app: gateway-gravité # Sélectionne le pod gateway-gravité
  policyTypes:
    - Egress
  egress:
    - to:
        - ipBlock:
            cidr: <adresse-IP-du-bridge>/32 # Remplacer par l'adresse IP du service bridge
      ports:
        - protocol: TCP
          port: <port-du-service-bridge> # Spécifiez le port utilisé par le bridge
