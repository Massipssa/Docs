// CreNumberCard.tsx

import React from 'react';
import { Typography, Chip } from '@mui/material';
import { LIGHT_GRAY_COLOR } from '@constants/color.constant';

interface CreNumberCardProps {
    filterStatus: string;
    count: number;
    status: string;
    title: string;
    clickColor: string;
    setFilterStatus: (status: string) => void;
}

const CreNumberCard: React.FC<CreNumberCardProps> = React.memo(
    ({ filterStatus, count, status, title, clickColor, setFilterStatus }) => {
        
        // Helper function to handle click event
        const handleClick = () => {
            setFilterStatus(filterStatus === status ? '' : status);
        };

        // Styles
        const chipStyles = {
            width: '100%',
            minHeight: '60px',
            cursor: 'pointer',
            backgroundColor: filterStatus === status ? clickColor : LIGHT_GRAY_COLOR,
            display: 'flex',
            justifyContent: 'center',
            alignItems: 'center',
            border: `1px solid ${clickColor}`,
            borderRadius: '2px',
            '&:hover': {
                backgroundColor: clickColor,
            },
        };

        return (
            <Chip
                sx={chipStyles}
                onClick={handleClick}
                label={
                    <>
                        <Typography variant="h6" color="textSecondary" align="center">
                            {title}
                        </Typography>
                        <Typography variant="h5" align="center">
                            {count}
                        </Typography>
                    </>
                }
            />
        );
    }
);

export default CreNumberCard;
